function loadSuggestions() {
    const tbody = document.getElementById('suggestions-tbody');
    tbody.innerHTML = '
';

    fetch('/api/smart/suggestions')
        .then(r => r.json())
        .then(data => {
            document.getElementById('suggestions-count').textContent = data.suggestions.length;

            if (data.suggestions.length === 0) {
                tbody.innerHTML = '暂无建议';
                return;
            }

            tbody.innerHTML = data.suggestions.map(s => {
                const route = s.suggested_origin && s.suggested_destination
                    ? `${s.suggested_origin} → ${s.suggested_destination}`
                    : '-';
                const confidenceClass = s.confidence >= 0.8 ? 'badge-success' :
                                      s.confidence >= 0.5 ? 'badge-warning' : 'badge-error';

                return `
                    
                        ${s.invoice.split('/').pop()}
                        ${s.suggested_type || '-'}
                        ${route}
                        ${(s.confidence * 100).toFixed(0)}%
                        ${s.reason}
                    
                `;
            }).join('');
        })
        .catch(err => {
            tbody.innerHTML = '加载失败';
        });
}

function loadAssociations() {
    const contentDiv = document.getElementById('associations-content');
    contentDiv.innerHTML = '
';

    fetch('/api/smart/associate')
        .then(r => r.json())
        .then(data => {
            document.getElementById('associations-count').textContent = data.associations.length;

            if (data.associations.length === 0) {
                contentDiv.innerHTML = '
暂无可关联的发票
';
                return;
            }

            let html = '
';

            data.associations.forEach(assoc => {
                const tripName = assoc.trip.split('/')[-1].replace(/_/g, ' ');
                html += `
                    

                        
📁 ${tripName}

                        

                            可关联 ${assoc.count} 张发票
                        

                        

                            ${assoc.invoices.map(inv => '
• ' + inv.split('/').pop() + '
').join('')}
                        

                    

                `;
            });

            html += '
';
            contentDiv.innerHTML = html;
        })
        .catch(err => {
            contentDiv.innerHTML = '
加载失败
';
        });
}

function generateReport() {
    if (!learnedPatterns) {
        alert('请先执行"学习历史模式"');
        return;
    }

    const reportCard = document.getElementById('report-card');
    const reportContent = document.getElementById('report-content');
    reportCard.style.display = 'block';

    reportContent.textContent = '生成中...';

    fetch('/api/smart/report', {
        method: 'POST',
        headers: {'Content-Type': 'application/json'}
    })
    .then(r => r.json())
    .then(data => {
        if (data.success) {
            reportContent.textContent = data.content;
        } else {
            reportContent.textContent = '生成失败: ' + data.error;
        }
    })
    .catch(err => {
        reportContent.textContent = '生成失败';
    });
}

// Auto-load on page load if patterns exist

        })
